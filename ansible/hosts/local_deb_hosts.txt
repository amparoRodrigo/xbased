# Example inventory for X-Road installation from local .deb packages
# This inventory is for Ubuntu/Debian systems only

[lxd_servers]
localhost ansible_connection=lxd

[cs_servers]
cs.example.com

[ss_servers]
ss1.example.com
ss2.example.com

[cp_servers]

[ca_servers]

local_deb_source_path=/home/ubuntu/X-Road/pacotes-atualizados
# Variant selection
variant=vanilla
